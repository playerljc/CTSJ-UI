初始化


import DrafFactory from '@ctmobile/ui-droppable';
import '@ctmobile/ui-droppable/droppable.less';
const droppable = DrafFactory.create(document.getElementById('container'), {
  onPutSuccess(params) {
    const cloneEl = params.sourceEl;
    cloneEl.style.position = 'fixed';
    cloneEl.style.zIndex = '9999';
    cloneEl.style.left = `${params.rect.left}px`;
    cloneEl.style.top = `${params.rect.top}px`;
    const targetEls = params.targetEls;
    targetEls[0].appendChild(cloneEl);
    drag.refresh();
    return true;
  },
  dragSourceExtendClasses: ['sourceActive'],
  dragTargetExtendClasses: ['targetActive'],
  isDragSourceDisplay: true,
  isDragSourceExist: true,
  noDragReturnAnimate: true,
});
名称 说明
{HtmlElement} - el 父容器元素
{object} - config

  • 指示器配置 {object} - config

    dragSourceExtendClasses {Array<String>} - 拖动对象的附加样式，拖动移动起来后触发

    dragTargetExtendClasses {Array<String>} - 可放置对象的附加样式，当拖动到可以放置的区域时触发

    isDragSourceDisplay {Boolean} - 拖动后原始节点是否显示

    isDragSourceExist {Boolean} - 拖动之后原始节点是否存在

    noDragReturnAnimate {Boolean} - 不可放置的时候松开是否有动画返回效果

    onPutSuccess {Function} - 可以放置时的处理

    onDragClone {Function} - 自定义拖动元素

    onSourceEnter {Function} - source节点进入

    onSourceLeave {Function} - source节点离开
